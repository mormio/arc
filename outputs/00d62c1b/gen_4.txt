def transform(input: List[List[int]]) -> List[List[int]]:
    # Find the number of columns
    num_cols = len(input[0])
    
    # Create a new grid with the same number of rows and columns
    new_grid = [[0] * num_cols for _ in range(len(input))]
    
    # Iterate over each cell in the input grid
    for i in range(len(input)):
        for j in range(num_cols):
            # If the cell is 3, increment the count of 3s in the corresponding column
            if input[i][j] == 3:
                new_grid[i][j] = input[i][j]
            else:
                # Find the count of 3s in the corresponding column
                count = sum(1 for k in range(len(input)) if input[k][j] == 3)
                new_grid[i][j] = count
    
    return new_grid
